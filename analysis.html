# --- Frequency""---
# Frequency""""""
def f_score(x):
    if x <= 1:
        return 1
    elif x <= 3:
        return 2
    elif x <= 5:
        return 3
    else:
        return 4

rfm_data['F'] = rfm_data['Frequency'].apply(f_score)
print("- Frequency")

# --- MonetaryValue ---
# 4
m_labels = range(1, 5)
